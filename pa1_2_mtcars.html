p_box_cat = (
    ggplot(df, aes(x="am", y="mpg"))

Answer (Boxplots of mpg by am — differences) - With am as text: two clear boxes (0 vs 1) - With am as float: x-axis is numeric and the grouping looks weird - Boxplots work better when am is text

df_lollipop = df.sort_values("mpg", ascending=False).head(10).copy()
p_lollipop = (
    ggplot(df_lollipop, aes(x="name", y="mpg"))
    + geom_segment(aes(x="name", xend="name", y=0, yend="mpg"))
    + geom_point(size=2.5)
    + coord_flip()
    + theme_bw()
    + labs(title="Top 10 Cars by MPG (Lollipop)",
           x="Car", y="MPG")
    + theme(axis_text_y=element_text(ha="right"))
)
p_lollipop
